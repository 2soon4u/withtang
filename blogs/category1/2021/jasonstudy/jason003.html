    package cn.edu.springmvc.test;
import org.springframework.web.servlet.mvc.Controller;
import org.springframework.web.context.WebApplicationContext;
import org.springframework.web.context.support.WebApplicationContextUtils;
import org.springframework.web.servlet.ModelAndView;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

public class MvcHelloController implements Controller{
	@Override
	public ModelAndView handleRequest(HttpServletRequest request, HttpServletResponse response)
	throws Exception{
		ModelAndView modelAndView = new ModelAndView();
		modelAndView.setViewName("hello"); //è®¾ç½®è§†å›¾å
		
		WebApplicationContext spring = 
				WebApplicationContextUtils.getWebApplicationContext(
						request.getSession().getServletContext());
                        //get bean from springmvc.xml
		
		User user = (User)spring.getBean("user");
		
		modelAndView.addObject(user);
		
		return modelAndView;
		
	}
	
}
